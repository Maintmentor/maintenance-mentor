CREATE TABLE repair_logs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  user_id UUID REFERENCES auth.users(id),
  category VARCHAR(20),  -- e.g., 'plumbing'
  unit VARCHAR(50),      -- e.g., 'Unit 204B'
  symptoms TEXT,
  steps JSONB,           -- Array of steps
  parts_needed JSONB,    -- Array of parts
  notes TEXT,            -- Optional tech notes
  status VARCHAR(20) DEFAULT 'completed'  -- e.g., 'completed', 'escalated'
);

-- Row Level Security (authenticated users only)
CREATE POLICY "Enable insert for authenticated users" ON repair_logs FOR INSERT WITH CHECK (auth.role() = 'authenticated');
CREATE POLICY "Enable read for authenticated users" ON repair_logs FOR SELECT USING (auth.role() = 'authenticated');
